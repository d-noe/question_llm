from questionnaire import *
from questionnaire.utils.pct_viz import *
import pandas as pd


pct = LikertQuestionnaire.from_json(
    PATH_LLM_QUEST+"data/pct.json"
)

fig, ax = make_pct_frame(size=10)

names = ["Karl Marx", "Donald Trump", "Joseph Stalin", "Milton Friedman"]
for n in names:
    num_answers = list(pd.read_csv(f"{RES_PATH}{'_'.join(n.lower().split(' '))}_pct.csv", sep="|")["numerical"].values)
    parsed_inputs = [
        {
            k: int(str(input)==k.lower())
            for k in keys
        }
        for keys, input in zip(
            pct.get_choices_keys(),
            num_answers
        )
    ]
    results = pct.evaluate(parsed_inputs)

    ax=place_image(
        im_path=f"{PATH_LLM_QUEST}data/images/political_figures/{n.split(' ')[1].lower()}.png",
        x_pos=results["economic"], y_pos=results["social"],
        ax=ax,
        zoom = .1
    )


plt.tight_layout()
plt.show()
